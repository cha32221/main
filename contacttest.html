const url = 'https://lassele.com'; // 여기에 접속할 웹사이트 URL을 입력하세요 const proxyUrl = 'https://cors-anywhere.herokuapp.com/'; // CORS 프록시 URL function checkWebsite() { fetch(proxyUrl + url) .then(response => { if (response.ok) { return response.text(); // 응답 본문을 읽어들입니다. } else { throw new Error(`상태 코드 ${response.status}`); } }) .then(data => { const resultText = `[${new Date().toLocaleString()}] ${url} 접속 성공: 응답 본문 크기 ${data.length} 바이트`; document.getElementById('result').innerHTML += `

${resultText}

`; }) .catch(error => { const resultText = `[${new Date().toLocaleString()}] ${url} 접속 실패: ${error.message}`; document.getElementById('result').innerHTML += `

${resultText}

`; }); } setInterval(checkWebsite, 60000); checkWebsite();
